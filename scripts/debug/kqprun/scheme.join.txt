CREATE OBJECT mysql_local_password (TYPE SECRET) WITH (value = "password");

CREATE EXTERNAL DATA SOURCE mysql_external_datasource WITH (
    SOURCE_TYPE="MySQL",
    LOCATION="mysql:3306",
    AUTH_METHOD="BASIC",
    LOGIN="root",
    DATABASE_NAME="fq",
    PASSWORD_SECRET_NAME="mysql_local_password"
);
